// run image
docker run image
docker run image ls 
docker run image echo 'something'

// get all containers
docker ps
docker ps --all

//create and start
docker create image
docker start image
docker start -a image
docker start -a container_id

//stop or get logs 
docker system prune  
docker logs container_id
docker stop container_id
docker kill container_id

//execute a command within a running Docker container
docker exec --it container_id command
docker exec --it container_id redis-cli

//shell
docker exec -it container_id sh
docker run -it image sh

//creating custom image
docker build .

//tagging image
docker build -t thesystemsmith/redis:latest . 
docker run docker.io/thesystemsmith/redis   

//buildkit disable 
DOCKER_BUILDKIT=0 docker build .

//port mapping
docker run -p 8080<incoming> : 8080<container_port> image

//docker-compose
docker-compose up 
docker-compose up --build //if anything is changes in the container

//start and stop
docker-compose up -d    
docker-compose down
